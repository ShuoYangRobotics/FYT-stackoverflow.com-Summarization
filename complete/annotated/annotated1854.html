Apart from patching the system javaos.py ( http : //osdir.com/ml/lang.jython.devel/2006-08/msg00035.html ) ( which I ca n't do , I have to use weblogic out of the box ) , this is what I use : 

def iswindows():
  os = java.lang.System.getProperty( "os.name" )
  return os.lower().find("win") > -1

/usr/bin/python3.2 

def cls():
    from subprocess import call
    from platform import system

    os = system()
    if os == 'Linux':
        call('clear', shell = True)
    elif os == 'Windows':
        call('cls', shell = True)
